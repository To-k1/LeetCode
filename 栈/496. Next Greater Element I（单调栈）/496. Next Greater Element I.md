```python
class Solution:
    def nextGreaterElement(self, nums1: List[int], nums2: List[int]) -> List[int]:
        ans=[-1]*len(nums1)
        stack=[]
        index={}
        for i in nums2:
            #python中空列表在条件判断中会转化为bool的0
            while stack and i>stack[-1]:
                index[stack[-1]]=i
                stack.pop()
            stack.append(i)
        for i in range(len(nums1)):
            ans[i]=index.get(nums1[i])
            if not ans[i]:
                ans[i]=-1
        return ans
```
## C++
```C++
class Solution {
public:
    vector<int> nextGreaterElement(vector<int>& nums1, vector<int>& nums2) {
        auto n1=nums1.size();
        auto n2=nums2.size();
        vector<int> ans(n1,-1);
        map<int,int> index;
        stack<int> s;
        for(auto i:nums2){
            //单调栈，栈顶到栈底严格不递减
            while(!s.empty() && i>s.top()){
                index[s.top()]=i;
                //用map实现
                //key=s.top(),value=i
                //保存了num2中s.top()对应的数对应的下一个更大元素值
                s.pop();
            }
            s.push(i);
        }
        //此时index保存了num2每一个元素的下一个更大元素值
        for(int i=0;i!=n1;++i){
            auto iter=index.find(nums1[i]);
            iter==index.end() ? :ans[i]=iter->second;//first,second表示key,value
        }
        return ans;
    }
};
```
